fn main() {
    let a = 34.0;
    let b = 80.0;

    println!("{} + {} = {}", a, b, a + b);
    println!("{} - {} = {}", b, a, b - a); // To avoid negative result
    println!("{} * {} = {}", a, b, a * b);
    println!("{} / {} = {}", b, a, b / a); // Integer division
    println!("{} % {} = {}", b, a, b % a); // Remainder
}
